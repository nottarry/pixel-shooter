html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Pixel Shooter</title>
  <style>
    body {
      margin: 0;
      background: black;
      overflow: hidden;
    }
    canvas {
      display: block;
      margin: auto;
      background: #111;
      image-rendering: pixelated;
    }
  </style>
</head>
<body>
<canvas id="gameCanvas" width="800" height="600"></canvas>
<script>
const canvas = document.getElementById("gameCanvas");
const ctx = canvas.getContext("2d");

const keys = {};
document.addEventListener("keydown", e => keys[e.code] = true);
document.addEventListener("keyup", e => keys[e.code] = false);

class Entity {
  constructor(x, y, size, color) {
    this.x = x;
    this.y = y;
    this.size = size;
    this.color = color;
    this.vx = 0;
    this.vy = 0;
    this.health = 100;
  }
  draw() {
    ctx.fillStyle = this.color;
    ctx.fillRect(this.x, this.y, this.size, this.size);
  }
}

class Bullet {
  constructor(x, y, vx, vy, damage) {
    this.x = x;
    this.y = y;
    this.vx = vx;
    this.vy = vy;
    this.size = 6;
    this.damage = damage;
  }
  update() {
    this.x += this.vx;
    this.y += this.vy;
  }
  draw() {
    ctx.fillStyle = "yellow";
    ctx.fillRect(this.x, this.y, this.size, this.size);
  }
}

class PowerUp {
  constructor(x, y, type) {
    this.x = x;
    this.y = y;
    this.size = 12;
    this.type = type; // "health" or "damage"
  }
  draw() {
    ctx.fillStyle = this.type === "health" ? "lime" : "red";
    ctx.fillRect(this.x, this.y, this.size, this.size);
  }
}

const player = new Entity(400, 300, 20, "cyan");
let bullets = [];
let enemies = [];
let powerups = [];
let score = 0;
let bulletDamage = 20;

function spawnEnemy() {
  let side = Math.floor(Math.random() * 4);
  let x, y;
  if (side === 0) { x = Math.random() * canvas.width; y = -20; }
  if (side === 1) { x = Math.random() * canvas.width; y = canvas.height + 20; }
  if (side === 2) { x = -20; y = Math.random() * canvas.height; }
  if (side === 3) { x = canvas.width + 20; y = Math.random() * canvas.height; }
  let enemy = new Entity(x, y, 20, "orange");
  enemy.health = 50;
  enemies.push(enemy);
}

function spawnPowerUp() {
  let type = Math.random() > 0.5 ? "health" : "damage";
  let x = Math.random() * (canvas.width - 20);
  let y = Math.random() * (canvas.height - 20);
  powerups.push(new PowerUp(x, y, type));
}

function gameLoop() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // Movement
  if (keys["KeyW"]) player.y -= 4;
  if (keys["KeyS"]) player.y += 4;
  if (keys["KeyA"]) player.x -= 4;
  if (keys["KeyD"]) player.x += 4;

  // Shooting
  if (keys["Space"] && Date.now() % 10 < 2) {
    bullets.push(new Bullet(player.x + player.size/2, player.y, 0, -6, bulletDamage));
  }

  // Update bullets
  bullets.forEach((b, i) => {
    b.update();
    b.draw();
    if (b.x < 0 || b.y < 0 || b.x > canvas.width || b.y > canvas.height) {
      bullets.splice(i, 1);
    }
  });

  // Update enemies
  enemies.forEach((e, i) => {
    let dx = player.x - e.x;
    let dy = player.y - e.y;
    let dist = Math.hypot(dx, dy);
    e.vx = (dx / dist) * 2;
    e.vy = (dy / dist) * 2;
    e.x += e.vx;
    e.y += e.vy;
    e.draw();

    // Check collision with player
    if (Math.abs(e.x - player.x) < 20 && Math.abs(e.y - player.y) < 20) {
      player.health -= 0.3;
    }

    // Bullet hit
    bullets.forEach((b, j) => {
      if (Math.abs(b.x - e.x) < 20 && Math.abs(b.y - e.y) < 20) {
        e.health -= b.damage;
        bullets.splice(j, 1);
      }
    });

    if (e.health <= 0) {
      enemies.splice(i, 1);
      score += 10;
      if (Math.random() < 0.2) spawnPowerUp();
    }
  });

  // Draw player
  player.draw();

  // Power-ups
  powerups.forEach((p, i) => {
    p.draw();
    if (Math.abs(p.x - player.x) < 20 && Math.abs(p.y - player.y) < 20) {
      if (p.type === "health") player.health = Math.min(100, player.health + 30);
      if (p.type === "damage") bulletDamage += 10;
      powerups.splice(i, 1);
    }
  });

  // HUD
  ctx.fillStyle = "white";
  ctx.font = "16px monospace";
  ctx.fillText("Health: " + Math.floor(player.health), 10, 20);
  ctx.fillText("Score: " + score, 10, 40);

  if (player.health <= 0) {
    ctx.fillStyle = "red";
    ctx.font = "40px monospace";
    ctx.fillText("GAME OVER", canvas.width/2 - 120, canvas.height/2);
    return;
  }

  requestAnimationFrame(gameLoop);
}

// Spawn enemies periodically
setInterval(spawnEnemy, 2000);

gameLoop();
</script>
</body>
</html>
